<statement> 	   -> <if_statement> | <for_statement> | <while_statement> | <do_while_statement> | <assignment_statement>

<if_statement>       -> "if" <expression_statement> ":" <statements> <elif_statement>* <else_statement>?

<elif_statement>     -> "elif" <expression_statement> ":" <statements> | epslion

<else_statement>     -> "else" ":" <statements>

<for_statement> 	   -> "for" <name_identifier> "in" <expression_statement> ":" <statements> <else_statement>?

<while_statement>    -> "while" <expression_statement> ":" <statements> <else_statement>?

<do_while_statement> -> "while" ":" <statements> "until" <expression_statement>

<statements> 	         -> <statement> <statements> | epslion

<assignment_statement> 	   -> <name_identifier> <assignment_operator> <expression_statement>

<expression_statement> 	   -> <term> | <expression_statement> <add_operator> <term>

<term> 		   -> <factor> | <term> <mult_operator> <factor>

<factor> 		   -> <name> | <number_identifier> | "(" <expression_statement> ")" | <unary_operator> <factor>

<name_identifier> 		   -> [a-zA-Z_][a-zA-Z0-9_]*

<number_identifier> 		   -> [0-9]+

<add_operator> 	   -> "+" | "-"

<mult_operator>      -> "*" | "/" | "%"

<unary_operator>     -> "+" | "-"
